                sendClickRequest(clickId);
            }

            console.error("Invalid tokens received:", tokens);
        }
    })

    function sendClickRequest(clickId) {
        if (!!clickId) {
            fetch("api/click", {
                method: "POST",
                body: JSON.stringify({
                    clickid: clickId,
                    page: window.location.pathname
                }),
                headers: {"Content-type": "application/json; charset=UTF-8"}
            });
        }
    }

    function sendClickFbRequest(clickId, fbp, fbclid) {
        if (!!clickId) {
            fetch("api/fbclick", {
                method: "POST",

                    clickid: clickId,
                    fbp: fbp,
                    fbclid: fbclid,
                    sourceUrl: window.location.href
                }),
                headers: {"Content-type": "application/json; charset=UTF-8"}
